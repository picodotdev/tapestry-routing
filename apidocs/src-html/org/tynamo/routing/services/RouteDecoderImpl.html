001package org.tynamo.routing.services;
002
003import org.apache.tapestry5.EventContext;
004import org.apache.tapestry5.SymbolConstants;
005import org.apache.tapestry5.internal.EmptyEventContext;
006import org.apache.tapestry5.internal.URLEventContext;
007import org.apache.tapestry5.ioc.annotations.Symbol;
008import org.apache.tapestry5.services.*;
009import org.slf4j.Logger;
010import org.tynamo.routing.Route;
011
012import java.util.regex.Matcher;
013
014public class RouteDecoderImpl implements RouteDecoder {
015
016        private static final char SLASH = '/';
017
018        private final LocalizationSetter localizationSetter;
019        private final boolean encodeLocaleIntoPath;
020        private final String applicationFolder;
021
022        private final URLEncoder urlEncoder;
023        private final ContextValueEncoder valueEncoder;
024        private final Logger logger;
025
026        public RouteDecoderImpl(LocalizationSetter localizationSetter,
027                                @Symbol(SymbolConstants.ENCODE_LOCALE_INTO_PATH) boolean encodeLocaleIntoPath,
028                                @Symbol(SymbolConstants.APPLICATION_FOLDER) final String applicationFolder,
029                                URLEncoder urlEncoder,
030                                ContextValueEncoder valueEncoder,
031                                Logger logger) {
032
033
034                this.localizationSetter = localizationSetter;
035                this.encodeLocaleIntoPath = encodeLocaleIntoPath;
036
037                this.applicationFolder = applicationFolder.equals("") ? "" : SLASH + applicationFolder;
038
039                this.urlEncoder = urlEncoder;
040                this.valueEncoder = valueEncoder;
041                this.logger = logger;
042        }
043
044        private String getLocaleFromPath(String path) {
045                if (!encodeLocaleIntoPath) return null;
046                // we have to get the possibly encoded locale from the request
047                // the following was copied and modified from AppPageRenderLinkTransformer.decodePageRenderRequest(...)
048                String[] split = path.substring(1).split("/");
049                if (split.length > 0 && !"".equals(split[0])) {
050                        String possibleLocaleName = split[0];
051                        // Might be just the page activation context, or it might be locale then page
052                        // activation context
053                        return localizationSetter.isSupportedLocaleName(possibleLocaleName) ? possibleLocaleName : null;
054                }
055                return null;
056        }
057
058        private String removeAppFolderAndLocaleFromPath(final Request request) {
059
060                String path = request.getPath();
061
062                if (this.applicationFolder.length() > 0) {
063                        path = path.substring(applicationFolder.length());
064                }
065
066                String locale = getLocaleFromPath(path);
067
068                if (locale != null) {
069                        localizationSetter.setLocaleFromLocaleName(locale);
070                        path = path.substring(locale.length() + 1);
071                }
072                return path.length() > 1 && path.charAt(path.length() - 1) == '/' ? path.substring(0, path.length() - 1) : path;
073        }
074
075        @Override
076        public PageRenderRequestParameters decodePageRenderRequest(final Route route, final Request request) {
077
078                Matcher matcher = route.getPattern().matcher(removeAppFolderAndLocaleFromPath(request));
079                if (!matcher.matches()) return null;
080
081                EventContext context;
082
083                int groupsSize = matcher.groupCount();
084
085                if (groupsSize < 1) {
086                        context = new EmptyEventContext();
087                } else {
088
089                        String[] split = new String[groupsSize];
090
091                        for (int i = 0; i < groupsSize; i++) {
092                                String value = matcher.group(i + 1);
093                                split[i] = urlEncoder.decode(value);
094                        }
095
096                        context = new URLEventContext(valueEncoder, split);
097                }
098
099                if (logger.isDebugEnabled())
100                        logger.debug("decoding render request using route: \"" + route.getPathExpression() + "\" for page: " + route.getCanonicalizedPageName());
101
102                return new PageRenderRequestParameters(route.getCanonicalizedPageName(), context, false);
103
104        }
105}
