001package org.tynamo.routing.services;
002
003import org.apache.tapestry5.internal.InternalConstants;
004import org.apache.tapestry5.internal.services.PageSource;
005import org.apache.tapestry5.ioc.annotations.PostInjection;
006import org.apache.tapestry5.ioc.annotations.Symbol;
007import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
008import org.apache.tapestry5.ioc.internal.util.Orderer;
009import org.apache.tapestry5.ioc.services.ClassNameLocator;
010import org.apache.tapestry5.services.ComponentClassResolver;
011import org.apache.tapestry5.services.ComponentClasses;
012import org.apache.tapestry5.services.InvalidationEventHub;
013import org.apache.tapestry5.services.InvalidationListener;
014import org.slf4j.Logger;
015import org.tynamo.routing.Route;
016import org.tynamo.routing.RoutingSymbols;
017
018import java.util.Collection;
019import java.util.Collections;
020import java.util.List;
021import java.util.Map;
022
023public class AnnotatedPagesManagerImpl implements AnnotatedPagesManager, InvalidationListener {
024
025        private final Collection<Class> contributedClasses;
026
027        private final Logger logger;
028        private final ClassNameLocator classNameLocator;
029        private final ComponentClassResolver componentClassResolver;
030        private final PageSource pageSource;
031        private final String appPackage;
032        private final Boolean preventScan;
033
034        private Orderer<Route> routeOrderer;
035        private List<Route> routes;
036        private Map<String, Route> routeMap;
037
038        private boolean objectWasInvalidated;
039
040        public AnnotatedPagesManagerImpl(Collection<Class> pages,
041                                         Logger logger,
042                                         ClassNameLocator classNameLocator,
043                                         ComponentClassResolver componentClassResolver,
044                                         PageSource pageSource,
045                                         @Symbol(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM) String appPackage,
046                                         @Symbol(RoutingSymbols.DISABLE_AUTODISCOVERY) Boolean preventScan) {
047
048                this.contributedClasses = pages;
049
050                this.logger = logger;
051                this.classNameLocator = classNameLocator;
052                this.componentClassResolver = componentClassResolver;
053                this.pageSource = pageSource;
054
055                this.appPackage = appPackage;
056                this.preventScan = preventScan;
057
058                this.objectWasInvalidated = true;
059
060                routeOrderer = new Orderer<Route>(logger);
061                routeMap = CollectionFactory.newConcurrentMap();
062        }
063
064        @Override
065        public void add(Route route, String... order) {
066                routeOrderer.add(route.getCanonicalizedPageName(), route, order);
067                routeMap.put(route.getCanonicalizedPageName(), route);
068        }
069
070        private void eagerLoadPages() {
071
072                if (!preventScan) {
073                        for (String className : classNameLocator.locateClassNames(appPackage + "." + InternalConstants.PAGES_SUBPACKAGE)) {
074                                String pageName = componentClassResolver.resolvePageClassNameToPageName(className);
075                                try {
076                                        pageSource.getPage(componentClassResolver.canonicalizePageName(pageName));
077                                } catch (Exception e) {
078                                        // TYNAMO-222
079                                        logger.error(e.getMessage());
080                                }
081                        }
082                }
083
084                for (Class page : contributedClasses) {
085                        String pageName = componentClassResolver.resolvePageClassNameToPageName(page.getName());
086                        pageSource.getPage(componentClassResolver.canonicalizePageName(pageName));
087                }
088
089                objectWasInvalidated = false;
090        }
091
092        @PostInjection
093        public void listenForInvalidations(@ComponentClasses InvalidationEventHub hub) {
094                hub.addInvalidationListener(this);
095        }
096
097        @Override
098        public void objectWasInvalidated() {
099                objectWasInvalidated = true;
100                routeOrderer = new Orderer<Route>(logger);
101                routeMap.clear();
102                routes = null;
103        }
104
105        @Override
106        public Route getRoute(String canonicalizedPageName) {
107                if (objectWasInvalidated) eagerLoadPages();
108                return routeMap.get(canonicalizedPageName);
109        }
110
111        @Override
112        public List<Route> getRoutes() {
113                if (objectWasInvalidated) eagerLoadPages();
114                if (routes == null) {
115                        routes = Collections.unmodifiableList(routeOrderer.getOrdered());
116                }
117                return routes;
118        }
119}
