001package org.tynamo.routing.services;
002
003import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
004import org.tynamo.routing.Route;
005
006import java.util.Collections;
007import java.util.List;
008import java.util.Map;
009
010public class RouteProviderImpl implements RouteProvider {
011
012        private final List<Route> routes;
013        private final Map<String, Route> routeMap = CollectionFactory.newConcurrentMap();
014
015        public RouteProviderImpl(List<Route> routes) {
016                this.routes = Collections.unmodifiableList(routes);
017                for (Route route : routes) {
018                        routeMap.put(route.getCanonicalizedPageName(), route);
019                }
020        }
021
022        @Override
023        public Route getRoute(String canonicalizedPageName) {
024                return routeMap.get(canonicalizedPageName);
025        }
026
027        @Override
028        public List<Route> getRoutes() {
029                return routes;
030        }
031}
