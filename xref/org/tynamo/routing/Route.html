1   package org.tynamo.routing;
2   
3   import java.util.regex.Matcher;
4   import java.util.regex.Pattern;
5   
6   public class Route {
7   
8   	public static final char SLASH = '/';
9   
10  	private static final String URI_PARAM_NAME_REGEX = "\\w[\\w\\.-]*";
11  	private static final String URI_PARAM_REGEX_REGEX = "[^{}][^{}]*";
12  	private static final String URI_PARAM_REGEX = "\\{\\s*(" + URI_PARAM_NAME_REGEX + ")\\s*(:\\s*(" + URI_PARAM_REGEX_REGEX + "))?\\}";
13  	private static final Pattern URI_PARAM_PATTERN = Pattern.compile(URI_PARAM_REGEX);
14  
15  	private final String canonicalizedPageName;
16  	private final String pathExpression;
17  	private final Pattern pattern;
18  
19  	public Route(final String pathExpression, final String canonicalizedPageName) {
20  
21  		this.canonicalizedPageName = canonicalizedPageName;
22  
23  		// remove ending slash unless it's the root path
24  		this.pathExpression = pathExpression.length() > 1 && pathExpression.charAt(pathExpression.length() - 1) == SLASH ? pathExpression.substring(0, pathExpression.length() - 1) : pathExpression;
25  
26  		if (!(this.pathExpression.charAt(0) == SLASH)) {
27  			throw new RuntimeException(
28  					"ERROR: Expression: \"" + this.pathExpression + "\" in: \"" + canonicalizedPageName +
29  							"\" page should start with a \"/\"");
30  		}
31  
32  		String regex = buildExpression(this.pathExpression);
33  		pattern = Pattern.compile(regex);
34  
35  	}
36  
37  	static String buildExpression(String expression) {
38  
39  		String[] split = URI_PARAM_PATTERN.split(expression);
40  		Matcher withPathParam = URI_PARAM_PATTERN.matcher(expression);
41  		int i = 0;
42  		StringBuilder builder = new StringBuilder();
43  		if (i < split.length) builder.append(Pattern.quote(split[i++]));
44  
45  		while (withPathParam.find()) {
46  			String expr = withPathParam.group(3);
47  			builder.append("(");
48  			if (expr == null) {
49  				builder.append("[^/]+");
50  			} else {
51  				throw new RuntimeException("regular expression mappings are not yet supported");
52  			}
53  
54  			builder.append(")");
55  			if (i < split.length) builder.append(Pattern.quote(split[i++]));
56  		}
57  
58  		return builder.toString();
59  	}
60  
61  	public String getPathExpression() {
62  		return pathExpression;
63  	}
64  
65  	public String getCanonicalizedPageName() {
66  		return canonicalizedPageName;
67  	}
68  
69  	public Pattern getPattern() {
70  		return pattern;
71  	}
72  }
