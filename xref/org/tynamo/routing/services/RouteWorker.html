1   package org.tynamo.routing.services;
2   
3   import org.apache.tapestry5.model.MutableComponentModel;
4   import org.apache.tapestry5.plastic.PlasticClass;
5   import org.apache.tapestry5.services.ComponentClassResolver;
6   import org.apache.tapestry5.services.transform.ComponentClassTransformWorker2;
7   import org.apache.tapestry5.services.transform.TransformationSupport;
8   import org.tynamo.routing.annotations.At;
9   import org.tynamo.routing.annotations.Route;
10  
11  public class RouteWorker implements ComponentClassTransformWorker2 {
12  
13  	private final AnnotatedPagesManager annotatedPagesManager;
14  	private final ComponentClassResolver componentClassResolver;
15  	private final RouteFactory routeFactory;
16  
17  	public RouteWorker(AnnotatedPagesManager annotatedPagesManager, ComponentClassResolver componentClassResolver, RouteFactory routeFactory) {
18  		this.annotatedPagesManager = annotatedPagesManager;
19  		this.componentClassResolver = componentClassResolver;
20  		this.routeFactory = routeFactory;
21  	}
22  
23  	@Override
24  	public void transform(PlasticClass plasticClass, TransformationSupport support, MutableComponentModel model) {
25  
26  		String pathExpression = null;
27  		String[] order = {};
28  
29  		if (plasticClass.hasAnnotation(At.class)) {
30  			At ann = plasticClass.getAnnotation(At.class);
31  			pathExpression = ann.value();
32  			order = ann.order();
33  		} else if (plasticClass.hasAnnotation(Route.class)) {
34  			Route ann = plasticClass.getAnnotation(Route.class);
35  			pathExpression = ann.value();
36  			order = ann.order();
37  		}
38  
39  		if (pathExpression != null) {
40  			String pageName = componentClassResolver.resolvePageClassNameToPageName(plasticClass.getClassName());
41  			String canonicalized = componentClassResolver.canonicalizePageName(pageName);
42  
43  			org.tynamo.routing.Route route = routeFactory.create(pathExpression, canonicalized);
44  
45  			annotatedPagesManager.add(route, order);
46  		}
47  	}
48  }
