17  public class RouterLinkTransformer implements PageRenderLinkTransformer {
18  
19  	private static final char SLASH = '/';
20  
21  	private final RouteSource routeSource;
22  	private final Request request;
23  	private final RequestSecurityManager requestSecurityManager;
24  	private final Response response;
25  	private final ContextPathEncoder contextPathEncoder;
26  	private final BaseURLSource baseURLSource;
27  	private final String applicationFolder;
28  
29  	private static final int BUFFER_SIZE = 100;
30  	private PersistentLocale persistentLocale;
31  	private boolean encodeLocaleIntoPath;
32  
33  	public RouterLinkTransformer(RouteSource routeSource,
34  	                             Request request,
35  	                             RequestSecurityManager requestSecurityManager, Response response,
36  	                             ContextPathEncoder contextPathEncoder, BaseURLSource baseURLSource,
37  	                             PersistentLocale persistentLocale,
38  	                             @Symbol(SymbolConstants.ENCODE_LOCALE_INTO_PATH) boolean encodeLocaleIntoPath,
39  	                             @Symbol(SymbolConstants.APPLICATION_FOLDER) final String applicationFolder) {
40  		this.routeSource = routeSource;
41  		this.request = request;
42  		this.requestSecurityManager = requestSecurityManager;
43  		this.response = response;
44  		this.contextPathEncoder = contextPathEncoder;
45  		this.baseURLSource = baseURLSource;
46  		this.persistentLocale = persistentLocale;
47  		this.encodeLocaleIntoPath = encodeLocaleIntoPath;
48  		this.applicationFolder = applicationFolder;
49  	}

51  	public PageRenderRequestParameters decodePageRenderRequest(Request request) {
52  		// this transformer should not intervene in the request decoding process
53  		return null; // return null to continue down the PageRenderLinkTransformer chain
54  	}
55  
56  	public Link transformPageRenderLink(Link defaultLink, PageRenderRequestParameters parameters) {
57  
58  		String activePageName = parameters.getLogicalPageName();
59  
60  		Route route = routeSource.getRoute(activePageName);
61  
62  		if (route != null) {
63  			StringBuilder builder = new StringBuilder(BUFFER_SIZE);
64  
65  			if (!"".equals(request.getContextPath())) {
66  				// Build up the absolute URI.
67  				builder.append(request.getContextPath());
68  			}
69  
70  			encodeAppFolderAndLocale(builder);
71  
72  			// deal with the very special case of a simple SLASH as a path expression
73  			String pathExpression = route.getPathExpression();
74  			if (!(pathExpression.length() == 1 && pathExpression.charAt(0) == SLASH && !applicationFolder.equals(""))) {
75  				builder.append(MessageFormat.format(route.getPathExpression(), (Object[]) encode(parameters.getActivationContext())));
76  			}
77  
78  			Link link = new LinkImpl(builder.toString(), false, requestSecurityManager.checkPageSecurity(activePageName), response, contextPathEncoder, baseURLSource);
79  
80  			if (parameters.isLoopback())
81  				link.addParameter(TapestryConstants.PAGE_LOOPBACK_PARAMETER_NAME, "t");
82  
83  			return link;
84  		}
85  
86  		return null;
87  	}
88  
89  	private void encodeAppFolderAndLocale(StringBuilder builder)
90  	{
91  		if (!applicationFolder.equals(""))
92  		{
93  			builder.append(SLASH).append(applicationFolder);
94  		}
95  
96  		if (encodeLocaleIntoPath)
97  		{
98  			Locale locale = persistentLocale.get();
99  
100 			if (locale != null)
101 			{
102 				builder.append(SLASH);
103 				builder.append(locale.toString());
104 			}
105 		}
106 	}
107 
108 	private String[] encode(EventContext context) {
109 
110 		assert context != null;
111 		int count = context.getCount();
112 
113 		String output[] = new String[count];
114 
115 		for (int i = 0; i < count; i++) {
116 			Object raw = context.get(Object.class, i);
117 			output[i] = contextPathEncoder.encodeValue(raw);
118 		}
119 
120 		return output;
121 	}
122 }
