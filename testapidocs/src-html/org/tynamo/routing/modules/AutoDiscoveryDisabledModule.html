001package org.tynamo.routing.modules;
002
003import org.apache.tapestry5.ioc.Configuration;
004import org.apache.tapestry5.ioc.MappedConfiguration;
005import org.apache.tapestry5.ioc.annotations.Contribute;
006import org.apache.tapestry5.ioc.annotations.SubModule;
007import org.apache.tapestry5.ioc.services.ApplicationDefaults;
008import org.apache.tapestry5.ioc.services.SymbolProvider;
009import org.tynamo.routing.RoutingSymbols;
010import org.tynamo.routing.pages.SimplePage;
011import org.tynamo.routing.services.AnnotatedPagesManager;
012
013@SubModule(TestsModule.class)
014public class AutoDiscoveryDisabledModule {
015
016        @Contribute(SymbolProvider.class)
017        @ApplicationDefaults
018        public static void provideApplicationDefaults(MappedConfiguration<String, String> configuration) {
019                configuration.add(RoutingSymbols.DISABLE_AUTODISCOVERY, "true");
020        }
021
022        @Contribute(AnnotatedPagesManager.class)
023        public static void annotatedPagesManager(Configuration<Class> configuration) {
024                configuration.add(SimplePage.class);
025        }
026
027}
