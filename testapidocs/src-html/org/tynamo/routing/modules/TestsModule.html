001package org.tynamo.routing.modules;
002
003import org.apache.tapestry5.SymbolConstants;
004import org.apache.tapestry5.TapestryConstants;
005import org.apache.tapestry5.internal.InternalConstants;
006import org.apache.tapestry5.internal.InternalSymbols;
007import org.apache.tapestry5.ioc.MappedConfiguration;
008import org.apache.tapestry5.ioc.OrderedConfiguration;
009import org.apache.tapestry5.ioc.annotations.Contribute;
010import org.apache.tapestry5.ioc.annotations.Primary;
011import org.apache.tapestry5.ioc.services.ApplicationDefaults;
012import org.apache.tapestry5.ioc.services.SymbolProvider;
013import org.tynamo.routing.Route;
014import org.tynamo.routing.services.RouteFactory;
015import org.tynamo.routing.services.RouteProvider;
016
017public class TestsModule {
018
019        @Contribute(SymbolProvider.class)
020        @ApplicationDefaults
021        public static void provideApplicationDefaults(MappedConfiguration<String, Object> configuration) {
022                configuration.add(InternalConstants.TAPESTRY_APP_PACKAGE_PARAM, "org.tynamo.routing");
023                configuration.add(InternalSymbols.APP_NAME, "app");
024                configuration.add(InternalSymbols.APP_PACKAGE_PATH, "org/tynamo/routing");
025                configuration.add(SymbolConstants.SUPPORTED_LOCALES, "en_US,fi");
026                configuration.add(SymbolConstants.PRODUCTION_MODE, false);
027        }
028
029        @Primary @Contribute(RouteProvider.class)
030        public static void addRoutes(OrderedConfiguration<Route> configuration, RouteFactory routeFactory) {
031
032                String canonicalized = "subpackage/Unannotated";
033                configuration.add(canonicalized.toLowerCase(), routeFactory.create("/not/annotated/{0}", canonicalized));
034
035        }
036
037}
